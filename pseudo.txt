function main():
    connectionA = null
    connectionB = null
    try:
        connectionA = connect_to_database("databaseA")
        connectionB = connect_to_database("databaseB")
        while True:
            employee_data = fetch_employee_data(connectionA)
            sorted_employee_data = sort_employee_data(employee_data)
            copy_employee_data(sorted_employee_data, connectionB)
            if all_data_copied():
                print("All data copied. Exiting program.")
                break
            else:
                print("Not all data copied. Continuing...")
    catch SQLException as e:
        print(e.printStackTrace())
    finally:
        close_connections(connectionA, connectionB)
